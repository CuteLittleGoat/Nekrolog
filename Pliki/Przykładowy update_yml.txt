name: Update obituaries data

on:
  schedule:
    - cron: "*/30 * * * *"   # co 30 minut
  workflow_dispatch:

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Install deps
        run: pip install -r requirements.txt

      - name: Run collector
        run: python collector.py data/latest.json

      - name: Commit changes
        run: |
          git config user.name "obits-bot"
          git config user.email "obits-bot@users.noreply.github.com"
          git add data/latest.json
          git commit -m "Update latest.json" || echo "No changes"
          git push
